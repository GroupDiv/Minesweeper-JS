1 /**
  2  * sound.js
  3  * Handles the sound functionality of the game
  4  * The following sound class was borrowed from https://www.w3schools.com/graphics/game_sound.asp
  5  */
  6 class sound {
  7 	/**
  8 	 * @constructor
  9 	 * @param {*} src 
 10 	 */
 11 	constructor(src) {
 12 		this.sound = document.createElement("audio");
 13 		this.sound.src = src;
 14 		this.sound.setAttribute("preload", "auto");
 15 		this.sound.setAttribute("controls", "none");
 16 		this.sound.style.display = "none";
 17 		document.body.appendChild(this.sound);
 18 		this.play = function () {
 19 			this.sound.play();
 20 		};
 21 		this.stop = function () {
 22 			this.sound.pause();
 23 		};
 24 	}
 25 }
