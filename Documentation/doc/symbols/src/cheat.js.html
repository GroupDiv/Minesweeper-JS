  3     that create the canvas upon which the cheat information is displayed

  6 /**
  7  * Fills a graphics context which displays solution to the board.It is meant to have its display toggled on and off with the "Cheat Mode"
  8    button.
  9  * @pre A gameboard object and a correctly sized cheatContext graphical context have been generated
 10  * @param {*} board = the gameBoard passed from main
 11  * @param {*} cheatContext = the graphical context belonging to the canvas designated for cheat
 12  * @post cheatContext has been stylized and correctly populated with mine counts
 13  * @return None
 14  */
 15 function cheatGen(board, cheatContext) {
 16     let components = [];  // an empty, eventually 2d, array to add visual components too
 17     cheatContext.strokeStyle = 'rgb(153, 204, 255)';
 18     cheatContext.shadowColor = 'rgba(0,0,0,1)';
 19     cheatContext.fillStyle = 'rgb(255, 92, 51)';
 20 
 21     for(let i =0; i< board.width; i++){
 22         components[i] = [];
 23         for(let j =0; j< board.height;j++){
 24             let mineCount = board.nmines_array[i][j];
 25             components[i][j] = componentGen(i*50,j*50,cheatContext, mineCount);
 26         }
 27     }
 28 }
 29 
 30 /**
 31     Generates the graphical component (rectangle) corresponding to the passed coordinates
 32 
 33     @pre cheatContext exists but does not yet have a component at requested coordinates
 34     @param xy = coordinates (in pixels) of requested generation
 35     @param cheatContext = the graphical context belonging to the canvas designated for cheat
 36     @param mineCount = the number of adjacent mines to the tile being generated
 37     @post a recangle has been added to cheatContext at the requested location
 38     @return None
 39 */
 40 function componentGen(x,y,cheatContext, mineCount){
 41     let width = 50;
 42     let height = 50;
 43     cheatContext.font = "15px Georgia";
 44     if (mineCount == 9) {
 45         cheatContext.fillRect(x, y, width, height)
 46     }
 47     else {
 48         cheatContext.strokeRect(x, y, width, height);
 49         cheatContext.fillText(countCorrect(mineCount) ,x+20,y+30);
 50     }
 51 }
 52 
 53 /**
 54  * Fixes mineCount to prepare it the value to be printed in componentGen().  (All this does is keep cheat mode from being filled with a bunch of zeroes)
 55  * @pre none
 56  * @param mineCount = number of adjacent mines to a tile
 57  * @post mineCount is set to "" if a 0 was passed through
 58  * @return mineCout = an integer 1-8 (untouched from passed parameter) or the empty string ""
 59 */
 60 function countCorrect(mineCount){
 61     if (mineCount == 0) {
 62         return ("");
 63     }
 64     else {
 65         return mineCount;
 66     }
 67 }
