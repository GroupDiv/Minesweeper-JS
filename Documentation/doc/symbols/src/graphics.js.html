  2 /**  
  3  * @classdesc The game graphics are handled by the Grphics class, how the numbers are going to populate on the HTML/User side visual and inputs. 
  4 */
  5 class Graphics{
  6 	/**  
  7 	 * The default constructor takes no parameters and makes a Canvas (A game area) on the HTML to be populated.
  8 	*/
  9 	constructor(gameBoard){
 10 		this.comp = [];
 11 		this.canvas = document.createElement("canvas");
 12 		this.canvas.width = gameBoard.width*50;
 13 		this.canvas.height = gameBoard.height*50;
 14 		//this.canvas.style = "border: 1px solid #d3d3d3";
 15 		this.context = this.canvas.getContext("2d");
 16 		document.body.insertBefore(this.canvas, document.body.childNodes[0]);
 17 		this.nflags = gameBoard.nmines;
 18 		this.draw(gameBoard);	
 19 		
 20 		this.flagImg = new Image();
 21 		this.flagImg.src="flag.png";
 22 		}
 23 	/**  
 24 	 * Makes the specific components in the Canvas, i.e. the tiles for the game.
 25 	 * @param {Number} x - The x coordinate of the component to be made.
 26 	 * @param {Number} y - The y coordinate of the component to be made.
 27 	*/
 28 	component(x,y,t=""){
 29 
 30 
 31 		this.width = 50;
 32 		this.height = 50;
 33 		this.x = x;
 34 		this.y =y;
 35 		this.context.font = "15px Georgia";
 36 		this.context.strokeRect(this.x, this.y, this.width, this.height);
 37 		if (t == "F"){
 38 			this.context.drawImage(this.flagImg,x,y,50,50);
 39 			}
 40 		else{
 41 		this.context.fillText(t,x+20,y+30);
 42 		}
 43 	}
 44 
 45 	/**  
 46  	* This method makes all the components at the time when the game starts i.e. the game is initialized.
 47 	*/
 48 	draw(gameBoard){
 49 		this.context.strokeStyle = 'rgba(0,0,0,0.2)';
 50 		this.context.shadowColor = 'rgba(0,0,0,0.2)';
 51 		for(let i =0; i<gameBoard.width;i++){
 52 			this.comp[i] = [];
 53 			for(let j =0; j<gameBoard.height;j++){
 54 				this.comp[i][j] = this.component(i*50,j*50);
 55 			}
 56 		}
 57 	}
 58 }
